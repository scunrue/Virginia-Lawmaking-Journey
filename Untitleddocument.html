                });

            } else if (currentStage.type === 'vote') {

                mainButtonsContainer.style.display = 'none'; // Hide main buttons during automated vote process

                processVote(currentStage); // Automatically trigger vote

            } else if (currentStage.type === 'outcome') {

                resetButton.style.display = 'block';

                billStatusIndicator.classList.remove('bill-status-passed', 'bill-status-failed'); // Reset for outcome stage

                billStatusIndicator.textContent = currentStage.id === 'signed' ? '🎉' : '❌'; // Specific icons for final outcomes

                if (currentStage.id === 'signed') {

                     billStatusIndicator.classList.add('bill-status-passed');

                } else if (currentStage.id === 'vetoed') {

                    billStatusIndicator.classList.add('bill-status-failed');

                    billStatusIndicator.textContent = '🚫';

                } else {

                    billStatusIndicator.classList.add('bill-status-failed');

                }

            }

            // Reset indicator style for non-outcome stages

            if (currentStage.type !== 'outcome') {

                billStatusIndicator.classList.remove('bill-status-passed', 'bill-status-failed');

                billStatusIndicator.textContent = '📱'; // Default phone icon

            }

            messageBox.style.display = 'none'; // Hide message box by default

            messageBox.textContent = ''; // Clear message

        }

        // Function to handle moving to the next stage (for 'info' type stages)

        function nextStage() {

            currentStageIndex++;

            if (currentStageIndex >= stages.length) {

                currentStageIndex = stages.length - 1; // Prevent going out of bounds

            }

            updateDisplay();

        }

        // Function to handle choices made by the user in 'decision' stages

        function handleDecision(choice) {

            billSupport += choice.supportChange;

            // Clamp billSupport to a reasonable range to prevent extreme probabilities

            billSupport = Math.max(-3, Math.min(3, billSupport));

            showMessage(choice.message);

            updateSupportIndicator();

            // Move to the next stage which is typically a 'vote' stage

            const currentStage = stages[currentStageIndex];

            const nextStageId = currentStage.nextStageOnDecision;

            currentStageIndex = stages.findIndex(s => s.id === nextStageId);

            updateDisplay(); // Update display to show the vote stage

        }

        // Function to process a 'vote' stage based on billSupport

        function processVote(voteStage) {

            // Calculate success probability based on billSupport

            // Base chance of 50%, +/- 10% for each point of support (e.g., -3 = 20%, 0 = 50%, 3 = 80%)

            let successProbability = 0.5 + (billSupport * 0.1);

            successProbability = Math.max(0.1, Math.min(0.9, successProbability)); // Clamp probability between 10% and 90%

            // If a specific passThreshold is defined, modify probability

            // For example, if passThreshold is 1, a neutral billSupport (0) gives 50%, but needs higher for good chance

            if (billSupport < voteStage.passThreshold) {

                // If support is below threshold, significantly reduce success chance

                successProbability = Math.max(0.1, successProbability - 0.2); // Further penalty

            } else if (billSupport > voteStage.passThreshold) {

                 // If support is above threshold, significantly increase success chance

                successProbability = Math.min(0.9, successProbability + 0.2); // Further boost

            }

            const passed = Math.random() < successProbability;

            setTimeout(() => { // Small delay to allow message to be read

                if (passed) {

                    currentStageIndex = stages.findIndex(s => s.id === voteStage.nextPass);

                    billStatusIndicator.classList.add('bill-status-passed');

                    billStatusIndicator.textContent = '✅';

                    showMessage(voteStage.messageOnPass);

                } else {

                    currentStageIndex = stages.findIndex(s => s.id === voteStage.nextFail);

                    billStatusIndicator.classList.add('bill-status-failed');

                    billStatusIndicator.textContent = '❌';

                    showMessage(voteStage.messageOnFail);

                }

                updateDisplay(); // Update display to show the outcome of the vote

            }, 1500); // Wait 1.5 seconds before revealing vote outcome

        }

        function showMessage(msg) {

            messageBox.textContent = msg;

            messageBox.style.display = 'block';

        }

        // Event listeners

        nextButton.addEventListener('click', nextStage);

        resetButton.addEventListener('click', () => {

            currentStageIndex = 0;

            billSupport = 0; // Reset support

            updateDisplay();

            billStatusIndicator.classList.remove('bill-status-passed', 'bill-status-failed');

            billStatusIndicator.textContent = '📱';

            messageBox.style.display = 'none';

        });

        // Initialize the simulation

        updateDisplay();

    </script>
